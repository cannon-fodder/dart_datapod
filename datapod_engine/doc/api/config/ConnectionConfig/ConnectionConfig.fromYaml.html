factory ConnectionConfig.fromYaml(Map<dynamic, dynamic> yaml) {
  final Map<String, dynamic> attrs = {};
  yaml.forEach((key, value) {
    if (key != 'name') {
      attrs[key.toString()] = value;
    }
  });

  return ConnectionConfig(name: yaml['name'] as String, attributes: attrs);
}
