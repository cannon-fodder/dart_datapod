@override
DatabaseOperations<E, K> getOperations<E extends Object, K>() {
  final operations = _operationsByEntity[E];
  if (operations == null) {
    throw StateError(
        'No operations registered for entity type $E in this context.');
  }
  return operations as DatabaseOperations<E, K>;
}
